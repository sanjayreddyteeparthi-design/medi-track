  <script>
    // Splash logo logic
    window.addEventListener('DOMContentLoaded', function() {
      const splash = document.getElementById('splashLogoBg');
      const mainContent = document.getElementById('mainContent');
      setTimeout(() => {
        splash.classList.add('hide');
        setTimeout(() => {
          splash.style.display = 'none';
          mainContent.style.display = 'block';
        }, 500);
      }, 2000);
    });

    function initStorage() {
      if (!localStorage.getItem('users')) localStorage.setItem('users', JSON.stringify([]));
      if (!localStorage.getItem('appointments')) localStorage.setItem('appointments', JSON.stringify([]));
      if (!localStorage.getItem('documents')) localStorage.setItem('documents', JSON.stringify([]));
      if (!localStorage.getItem('notifications')) localStorage.setItem('notifications', JSON.stringify([]));
    }

    function showNotification(message) {
      const notification = document.getElementById('notification');
      document.getElementById('notification-message').textContent = message;
      notification.classList.remove('hidden');
      notification.classList.add('show');
      setTimeout(() => {
        notification.classList.remove('show');
        setTimeout(() => notification.classList.add('hidden'), 300);
      }, 3000);
    }

    function escapeHTML(str) {
      return str.replace(/[&<>"']/g, (m) => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;' })[m]);
    }

    function validateEmail(email) {
      return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
    }

    function showForm(formId) {
      ['home', 'loginForm', 'createAccountForm', 'dashboard', 'bookAppointmentForm', 'uploadDocumentForm', 'viewAppointments', 'viewDocuments', 'viewNotifications'].forEach(id => {
        document.getElementById(id).style.display = id === formId ? 'block' : 'none';
      });
      if (formId === 'viewAppointments') updateAppointmentsList();
      if (formId === 'viewDocuments') updateDocumentsList();
      if (formId === 'viewNotifications') updateNotificationsList();
      if (formId === 'dashboard') updateNotifications(); // Mock reminders
    }

    function backToHome() { 
      localStorage.removeItem('currentUser');
      showForm('home'); 
    }

    function logout() {
      localStorage.removeItem('currentUser');
      showForm('home');
    }

    function getCurrentUser() {
      return JSON.parse(localStorage.getItem('currentUser'));
    }

    function updateAppointmentsList() {
      const user = getCurrentUser();
      if (!user) return;
      const appointments = JSON.parse(localStorage.getItem('appointments') || '[]').filter(a => a.userEmail === user.email);
      const container = document.getElementById('appointmentsList');
      if (appointments.length === 0) {
        container.innerHTML = `<div class="text-center py-8 text-gray-500"><i class="fas fa-inbox text-4xl mb-3"></i><p>No appointments yet.</p></div>`;
        return;
      }
      container.innerHTML = '';
      appointments.forEach(app => {
        const div = document.createElement('div');
        div.className = 'item-card';
        div.innerHTML = `
          <h4 class="font-semibold text-lg">${escapeHTML(app.doctor)}</h4>
          <p>Date: ${escapeHTML(app.date)}</p>
          <p>Time: ${escapeHTML(app.time)}</p>
        `;
        container.appendChild(div);
      });
    }

    function updateDocumentsList() {
      const user = getCurrentUser();
      if (!user) return;
      const documents = JSON.parse(localStorage.getItem('documents') || '[]').filter(d => d.userEmail === user.email);
      const container = document.getElementById('documentsList');
      if (documents.length === 0) {
        container.innerHTML = `<div class="text-center py-8 text-gray-500"><i class="fas fa-inbox text-4xl mb-3"></i><p>No documents uploaded yet.</p></div>`;
        return;
      }
      container.innerHTML = '';
      documents.forEach(doc => {
        const div = document.createElement('div');
        div.className = 'item-card';
        div.innerHTML = `
          <h4 class="font-semibold text-lg">${escapeHTML(doc.doc_name)}</h4>
          <p>${escapeHTML(doc.doc_description || 'No description')}</p>
          ${doc.doc_file ? `<a href="${doc.doc_file}" target="_blank">View Document</a>` : ''}
        `;
        container.appendChild(div);
      });
    }

    function updateNotificationsList() {
      const user = getCurrentUser();
      if (!user) return;
      const notifications = JSON.parse(localStorage.getItem('notifications') || '[]').filter(n => n.userEmail === user.email);
      const container = document.getElementById('notificationsList');
      if (notifications.length === 0) {
        container.innerHTML = `<div class="text-center py-8 text-gray-500"><i class="fas fa-inbox text-4xl mb-3"></i><p>No notifications yet.</p></div>`;
        return;
      }
      container.innerHTML = '';
      notifications.forEach(notif => {
        const div = document.createElement('div');
        div.className = 'item-card';
        div.innerHTML = `<p>${escapeHTML(notif.message)}</p>`;
        container.appendChild(div);
      });
    }

    function updateNotifications() {
      const user = getCurrentUser();
      if (!user) return;
      const appointments = JSON.parse(localStorage.getItem('appointments') || '[]').filter(a => a.userEmail === user.email);
      const notifications = [];
      appointments.forEach(app => {
        const appDate = new Date(app.date);
        const today = new Date();
        if (appDate > today) {
          const diffDays = Math.ceil((appDate - today) / (1000 * 60 * 60 * 24));
          if (diffDays === 1) {
            notifications.push({ userEmail: user.email, message: `Reminder: Your appointment with ${app.doctor} is tomorrow at ${app.time}.` });
          }
        }
      });
      localStorage.setItem('notifications', JSON.stringify(notifications));
    }

    document.getElementById('loginForm').onsubmit = function(e) {
      e.preventDefault();
      const form = new FormData(this);
      const email = form.get('email');
      const password = form.get('password');
      if (!validateEmail(email)) {
        showNotification('Invalid email address.');
        return;
      }
      const users = JSON.parse(localStorage.getItem('users') || '[]');
      const user = users.find(u => u.email === email && u.password === password);
      if (user) {
        localStorage.setItem('currentUser', JSON.stringify(user));
        showNotification('Login successful!');
        showForm('dashboard');
      } else {
        showNotification('Invalid credentials.');
      }
    };

    document.getElementById('createAccountForm').onsubmit = function(e) {
      e.preventDefault();
      const form = new FormData(this);
      const name = form.get('name');
      const age = form.get('age');
      const email = form.get('email');
      const password = form.get('password');
      const confirmPassword = form.get('confirm_password');
      if (!validateEmail(email)) {
        showNotification('Invalid email address.');
        return;
      }
      if (password !== confirmPassword) {
        showNotification('Passwords do not match.');
        return;
      }
      const users = JSON.parse(localStorage.getItem('users') || '[]');
      if (users.find(u => u.email === email)) {
        showNotification('Email already exists.');
        return;
      }
      users.push({ name, age, email, password });
      localStorage.setItem('users', JSON.stringify(users));
      showNotification('Account created successfully! Please login.');
      showForm('loginForm');
    };

    document.getElementById('bookAppointmentForm').onsubmit = function(e) {
      e.preventDefault();
      const user = getCurrentUser();
      if (!user) return;
      const form = new FormData(this);
      const appointment = { ...Object.fromEntries(form), userEmail: user.email, timestamp: new Date().toISOString() };
      const arr = JSON.parse(localStorage.getItem('appointments') || '[]');
      arr.push(appointment);
      localStorage.setItem('appointments', JSON.stringify(arr));
      showNotification('Appointment booked successfully!');
      showForm('dashboard');
    };

    document.getElementById('uploadDocumentForm').onsubmit = function(e) {
      e.preventDefault();
      const user = getCurrentUser();
      if (!user) return;
      const form = new FormData(this);
      const file = form.get('doc_file');
      if (file && file.size > 0) {
        const reader = new FileReader();
        reader.onload = function(e) {
          const document = { ...Object.fromEntries(form), doc_file: e.target.result, userEmail: user.email, timestamp: new Date().toISOString() };
          const arr = JSON.parse(localStorage.getItem('documents') || '[]');
          arr.push(document);
          localStorage.setItem('documents', JSON.stringify(arr));
          showNotification('Document uploaded successfully!');
          showForm('dashboard');
        };
